## TERMINAL BASICS

## ANATOMY OF A COMMAND LINE THING

`COMMAND -[FLAG] ARGUMENT`

# CHANGE DIRECTORIES

`cd [DIRECTORY_NAME]`

# MAKE A DIRECTORY

`mkdir [DIRECTORY_NAME]`

# MAKE A NEW FILE

`touch [NAME_OF_FILE]`

# REMOVE A FILE

`rm -rf [NAME_OF_FILE]`

# LIST ALL CONTENTS

`ls -a [DIRECTORY_NAME]`

# MOVE A FILE

`mv [FROM] [TO]`

# COPY A FILE

`cp -r [FROM] [TO]`

# ACCES ROOT / GO HOME

`~` symbol for home
`cd ~` go home from anywhere

# PRINT ALL CONTENTS TO TERMINAL

`cat [NAME_OF_FILE]`

# WRITE TO A FILE

`echo "[CONTENT]" >> [NAME_OF_FILE]`


## GIT

## STATUS

`git status`

# COMMIT

`git commit -m "[COMMIT_MESSAGE]"`

# ADD

`git add [NAME_OF_FILE]`
`git add .` add all files

# PUSH

`git push [REMOTE] [BRANCH]`

# PULL

`git pull [REMOTE] [BRANCH]`

# HOW TO ADD A REMOTE

`git remote add [NAME_OF_REMOTE] [HTTP_LOCATION]`

# INITIALIZE A GIT REPO

`git init`

# CLONE

`git clone [HTTP_LOCATION]`

# LOG

`git log`
